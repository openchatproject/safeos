36 void validate_authority_precondition( const apply_context& context, const authority& auth ) {
37  for(const auto& a : auth.accounts) {
38  auto* acct = context.db.find<account_object, by_name>(a.permission.actor);
39  EOS_ASSERT( acct != nullptr, action_validate_exception,
40  "account '${account}' does not exist",
41  ("account", a.permission.actor)
42  );
43 
44  if( a.permission.permission == config::owner_name || a.permission.permission == config::active_name )
45  continue; // account was already checked to exist, so its owner and active permissions should exist
46 
47  if( a.permission.permission == config::eosio_code_name ) // virtual eosio.code permission does not really exist but is allowed
48  continue;
49 
50  try {

54  "permission '${perm}' does not exist",

57  }
