56  void add( uint64_t units, uint32_t ordinal, uint32_t window_size /* must be positive */ )
57  {
58  // check for some numerical limits before doing any state mutations
59  EOS_ASSERT(units <= max_raw_value, rate_limiting_state_inconsistent, "Usage exceeds maximum value representable after extending for precision");
60  EOS_ASSERT(std::numeric_limits<decltype(consumed)>::max() - consumed >= units, rate_limiting_state_inconsistent, "Overflow in tracked usage when adding usage!");
61 
62  auto value_ex_contrib = integer_divide_ceil(units * Precision, (uint64_t)window_size);
63  EOS_ASSERT(std::numeric_limits<decltype(value_ex)>::max() - value_ex >= value_ex_contrib, rate_limiting_state_inconsistent, "Overflow in accumulated value when adding usage!");
64 
65  if( last_ordinal != ordinal ) {
66  FC_ASSERT( ordinal > last_ordinal, "new ordinal cannot be less than the previous ordinal" );
67  if( (uint64_t)last_ordinal + window_size > (uint64_t)ordinal ) {
68  const auto delta = ordinal - last_ordinal; // clearly 0 < delta < window_size
69  const auto decay = make_ratio(
70  (uint64_t)window_size - delta,
71  (uint64_t)window_size
72  );
73 
74  value_ex = value_ex * decay;
75  } else {
76  value_ex = 0;
77  }
78 
79  last_ordinal = ordinal;
80  consumed = average();
81  }
82 
83  consumed += units;
84  value_ex += value_ex_contrib;
85  }
