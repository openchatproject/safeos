172  static account_name get_owner_for_identity( uint64_t receiver, identity_name ident ) {
173  // for each trusted owner certification
174  // check to see if the certification is still trusted
175  // check to see if the account has claimed it
176  certs_table certs( code, ident );
177  auto idx = certs.template get_index<N(bytuple)>();
178  auto itr = idx.lower_bound(certrow::key(N(owner), 1, 0));
179  account_name owner = 0;
180  while (itr != idx.end() && itr->property == N(owner) && itr->trusted) {
181  if (sizeof(account_name) == itr->data.size()) {
182  account_name account = *reinterpret_cast<const account_name*>(itr->data.data());
183  if (ident == get_claimed_identity(account)) {
184  if (is_trusted(itr->certifier) ) {
185  // the certifier is still trusted
186  if (!owner || owner == account) {
187  owner = account;
188  } else {
189  //contradiction found: different owners certified for the same identity
190  return 0;
191  }
192  } else if (DeployToAccount == receiver){
193  //the certifier is no longer trusted, need to unset the flag
194  idx.modify(itr, 0, [&](certrow& r) {
195  r.trusted = 0;
196  });
197  } else {
198  // the certifier is no longer trusted, but the code runs in read-only mode
199  }
200  }
201  } else {
202  // bad row - skip it
203  }
204  ++itr;
205  }
206  if (owner) {
207  //owner found, no contradictions among certifications flaged as trusted
208  return owner;
209  }
210  // trusted certification not found
211  // let's see if any untrusted certifications became trusted
212  itr = idx.lower_bound(certrow::key(N(owner), 0, 0));
213  while (itr != idx.end() && itr->property == N(owner) && !itr->trusted) {
214  if (sizeof(account_name) == itr->data.size()) {

216  if (ident == get_claimed_identity(account) && is_trusted(itr->certifier)) {
217  if (DeployToAccount == receiver) {

219  idx.modify(itr, 0, [&](certrow& r) {
