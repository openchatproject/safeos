34  // Orders permissions descending by weight, and breaks ties with Wait permissions being less than
35  // Key permissions which are in turn less than Account permissions
36  struct meta_permission_comparator {
37  bool operator()( const meta_permission& lhs, const meta_permission& rhs ) const {
38  get_weight_visitor scale;
39  auto lhs_weight = lhs.visit(scale);
40  auto lhs_type = lhs.which();
41  auto rhs_weight = rhs.visit(scale);
42  auto rhs_type = rhs.which();
43  return std::tie( lhs_weight, lhs_type ) > std::tie( rhs_weight, rhs_type );
44  }

183  bool satisfied( const AuthorityType& authority, permission_cache_type& cached_permissions, uint16_t depth ) {
184  // Save the current used keys; if we do not satisfy this authority, the newly used keys aren't actually used
185  auto KeyReverter = fc::make_scoped_exit([this, keys = _used_keys] () mutable {
186  _used_keys = keys;
187  });
188 
189  // Sort key permissions and account permissions together into a single set of meta_permissions
190  detail::meta_permission_set permissions;
191 
192  permissions.insert(authority.waits.begin(), authority.waits.end());
193  permissions.insert(authority.keys.begin(), authority.keys.end());
194  permissions.insert(authority.accounts.begin(), authority.accounts.end());
195 
196  // Check all permissions, from highest weight to lowest, seeing if provided authorization factors satisfies them or not
197  weight_tally_visitor visitor(*this, cached_permissions, depth);
198  for( const auto& permission : permissions )
199  // If we've got enough weight, to satisfy the authority, return!
200  if( permission.visit(visitor) >= authority.threshold ) {
201  KeyReverter.cancel();
202  return true;
203  }
204  return false;

238  auto status = authority_checker::permission_status_in_cache( cached_permissions, permission.permission );
239  if( !status ) {

241  bool r = false;
242  typename permission_cache_type::iterator itr = cached_permissions.end();
243 
244  bool propagate_error = false;
245  try {

247  propagate_error = true;
248  auto res = cached_permissions.emplace( permission.permission, being_evaluated );
249  itr = res.first;
250  r = checker.satisfied( std::forward<decltype(auth)>(auth), cached_permissions, recursion_depth + 1 );
251  } catch( const permission_query_exception& ) {
252  if( propagate_error )
253  throw;
254  else
255  return total_weight; // if the permission doesn't exist, continue without it
256  }
257 
258  if( r ) {

261  } else {
262  itr->second = permission_unsatisfied;
263  }
264  }
