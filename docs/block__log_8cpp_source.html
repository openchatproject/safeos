98  my->block_write = true;

100 
101  /* On startup of the block log, there are several states the log file and the index file can be
102  * in relation to each other.
103  *
104  * Block Log
105  * Exists Is New
106  * +------------+------------+
107  * Exists | Check | Delete |
108  * Index | Head | Index |
109  * File +------------+------------+
110  * Is New | Replay | Do |
111  * | Log | Nothing |
112  * +------------+------------+
113  *
114  * Checking the heads of the files has several conditions as well.
115  * - If they are the same, do nothing.
116  * - If the index file head is not in the log file, delete the index and replay.
117  * - If the index file head is in the log, but not up to date, replay from index head.
118  */
119  auto log_size = fc::file_size(my->block_file);
120  auto index_size = fc::file_size(my->index_file);
121 
122  if (log_size) {
123  ilog("Log is nonempty");
124  my->check_block_read();
125  my->block_stream.seekg( 0 );
126  uint32_t version = 0;
127  my->block_stream.read( (char*)&version, sizeof(version) );
128  FC_ASSERT( version > 0, "Block log was not setup properly with genesis information." );
129  FC_ASSERT( version == block_log::supported_version,
130  "Unsupported version of block log. Block log version is ${version} while code supports version ${supported}",
131  ("version", version)("supported", block_log::supported_version) );
132 
133  my->genesis_written_to_block_log = true; // Assume it was constructed properly.
134  my->head = read_head();
135  my->head_id = my->head->id();
136  edump((my->head->block_num()));
137 
138  if (index_size) {
139  my->check_block_read();
140  my->check_index_read();
141 
142  ilog("Index is nonempty");
143  uint64_t block_pos;
144  my->block_stream.seekg(-sizeof(uint64_t), std::ios::end);
145  my->block_stream.read((char*)&block_pos, sizeof(block_pos));
146 
147  uint64_t index_pos;
148  my->index_stream.seekg(-sizeof(uint64_t), std::ios::end);
149  my->index_stream.read((char*)&index_pos, sizeof(index_pos));
150 
151  if (block_pos < index_pos) {
152  ilog("block_pos < index_pos, close and reopen index_stream");
153  construct_index();
154  } else if (block_pos > index_pos) {
155  ilog("Index is incomplete");
156  construct_index();
157  }
158  } else {
159  ilog("Index is empty");
160  construct_index();
161  }
162  } else if (index_size) {
163  ilog("Index is nonempty, remove and recreate it");
164  my->index_stream.close();
165  fc::remove_all(my->index_file);
166  my->index_stream.open(my->index_file.generic_string().c_str(), LOG_WRITE);
167  my->index_write = true;
168  }
