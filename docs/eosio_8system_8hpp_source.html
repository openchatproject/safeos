112  static void on(const claimrewards& cr) {
113  require_auth(cr.owner);
114  eosio_assert(current_sender() == account_name(), "claimrewards can not be part of a deferred transaction");
115  producers_table producers_tbl(SystemAccount, SystemAccount);
116  auto prod = producers_tbl.find(cr.owner);
117  eosio_assert(prod != producers_tbl.end(), "account name is not in producer list");
118  eosio_assert(prod->active(), "producer is not active"); // QUESTION: Why do we want to prevent inactive producers from claiming their earned rewards?
119  if( prod->last_rewards_claim > 0 ) {
120  eosio_assert(now() >= prod->last_rewards_claim + seconds_per_day, "already claimed rewards within a day");
121  }
122  system_token_type rewards = prod->per_block_payments;
123  auto idx = producers_tbl.template get_index<N(prototalvote)>();
124  auto itr = --idx.end();
125 
126  bool is_among_payed_producers = false;
127  uint128_t total_producer_votes = 0;
128  uint32_t n = 0;
129  while( n < num_of_payed_producers ) {
130  if( !is_among_payed_producers ) {
131  if( itr->owner == cr.owner )
132  is_among_payed_producers = true;
133  }
134  if( itr->active() ) {
135  total_producer_votes += itr->total_votes;
136  ++n;
137  }
138  if( itr == idx.begin() ) {
139  break;
140  }
141  --itr;
142  }
143 
144  if (is_among_payed_producers && total_producer_votes > 0) {
145  if( global_state_singleton::exists() ) {
146  auto parameters = global_state_singleton::get();
147  auto share_of_eos_bucket = system_token_type( static_cast<uint64_t>( (prod->total_votes * parameters.eos_bucket.quantity) / total_producer_votes ) ); // This will be improved in the future when total_votes becomes a double type.
148  rewards += share_of_eos_bucket;
149  parameters.eos_bucket -= share_of_eos_bucket;
150  global_state_singleton::set(parameters);
151  }
152  }
153

157  p.last_rewards_claim = now();
158  p.per_block_payments.quantity = 0;
159  });
