18  // to its 5-bit slot starting with the highest slot for the first char.
19  // The 13th char, if str is long enough, is encoded into 4-bit chunk
20  // and placed in the lowest 4 bits. 64 = 12 * 5 + 4

22  {
23  uint64_t name = 0;

27  // expansion to uint64 type, the compilation fails at the point of usage
28  // of string_to_name(), where the usage requires constant (compile time) expression.
29  name |= (char_to_symbol(str[i]) & 0x1f) << (64 - 5 * (i + 1));
