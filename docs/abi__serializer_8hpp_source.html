424  EOS_ASSERT(vo.contains("compression"), packed_transaction_type_exception, "Missing compression");

429  if( vo.contains("packed_trx") && vo["packed_trx"].is_string() && !vo["packed_trx"].as_string().empty() ) {
430  from_variant(vo["packed_trx"], ptrx.packed_trx);
431  auto trx = ptrx.get_transaction(); // Validates transaction data provided.
432  if( vo.contains("packed_context_free_data") && vo["packed_context_free_data"].is_string() && !vo["packed_context_free_data"].as_string().empty() ) {
433  from_variant(vo["packed_context_free_data"], ptrx.packed_context_free_data );
434  } else if( vo.contains("context_free_data") ) {
435  vector<bytes> context_free_data;
436  from_variant(vo["context_free_data"], context_free_data);
437  ptrx.set_transaction(trx, context_free_data, ptrx.compression);
438  }
439  } else {

445  from_variant(vo["packed_context_free_data"], ptrx.packed_context_free_data );
446  context_free_data = ptrx.get_context_free_data();
447  } else if( vo.contains("context_free_data") ) {
448  from_variant(vo["context_free_data"], context_free_data);
