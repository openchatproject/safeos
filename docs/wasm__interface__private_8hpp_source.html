36 
37  for(const DataSegment& data_segment : module.dataSegments) {
38  FC_ASSERT(data_segment.baseOffset.type == InitializerExpression::Type::i32_const);
39  FC_ASSERT(module.memories.defs.size());
40  const U32 base_offset = data_segment.baseOffset.i32;
41  const Uptr memory_size = (module.memories.defs[0].type.size.min << IR::numBytesPerPageLog2);
42  if(base_offset >= memory_size || base_offset + data_segment.data.size() > memory_size)
43  FC_THROW_EXCEPTION(wasm_execution_error, "WASM data segment outside of valid memory range");
44  if(base_offset + data_segment.data.size() > mem_image.size())
45  mem_image.resize(base_offset + data_segment.data.size(), 0x00);
